	}
	
	selphereg_a_place_in_the_sun = {
		icon = mission_secure_khandesh
		required_missions = {  }
		position = 1
		
		provinces_to_highlight = {
			area_for_scope_province = {
				is_capital_of = ROOT
			}
			OR = {
				NOT = { owned_by = ROOT }
				AND = {
					is_capital_of = ROOT
					NOT = { development = 14 }
				}
			}
		}
		
		trigger = {
			capital_scope = {
				area_for_scope_province = {
					type = all

			}
		}
		
		effect = { 
			capital_scope = {
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
				area = {
					limit = { owned_by = ROOT }
					add_province_modifier = {
						name = "growth_of_capital"
						duration = 9125 #25 years
					}
				}
			}
			increase_legitimacy_medium_effect = yes
			add_prestige = 10
		}
	}

selpheregi4_missions = {
	slot = 4
	generic = no
	ai = yes 
	potential = {
		primary_culture = selphereg
		NOT = {
			tag = H01
			tag = H21
		}
	}
	
	selphereg_precursor_expeditons = {
		icon = mission_jawal_silver_mine
		required_missions = { }
		position = 1
		
		provinces_to_highlight = {
			OR = {
				AND = {
					NOT = { owned_by = ROOT }
					trade_goods = precursor_relics
					has_discovered = ROOT
				}
				AND = {
					owned_by = ROOT
					trade_goods = precursor_relics
					NOT = { base_production = 5 }
				}
			}
		}
		
		trigger = {
			precursor_relics = 2
			all_owned_province = {
				OR = {
					AND = {
						trade_goods = precursor_relics
						base_production = 5
					}
					NOT = { trade_goods = precursor_relics }
				}
			}
		}
		
		effect = {
			random_owned_province = {
				limit = { trade_goods = precursor_relics }
				add_base_manpower = 1
				add_base_production = 1
				add_permanent_province_modifier = {
					name = selphereg_precursor_excavation
					duration = 18250
				}
			}
		}
	}
	
	selphereg_renovate_arakeprun = {
		icon = mission_city_prosperity
		required_missions = {
			selphereg_precursor_expeditons
		}
		position = 2
		
		provinces_to_highlight = {
			province_id = 2007
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
				NOT = { has_production_building_trigger = yes }
			}
		}
		
		trigger = {
			owns_core_province = 2007
			precursor_relics = 2
			2007 = {
				has_production_building_trigger = yes
			}
			years_of_income = 2
		}
		
		effect = {
			2007 = {
				remove_province_modifier = arakeprun_precursor_tool_deprivation
			}
			add_years_of_income = -1.75
			add_prestige = 10
			hidden_effect = { set_country_flag = eordand_rebuilt_arakeprun }
		}
	}
}
