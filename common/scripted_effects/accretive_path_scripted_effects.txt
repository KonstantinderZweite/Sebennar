accretive_path_generic_chi_stacking_effect = {
	if = {
		limit = {
			religion = accretive_path
			NOT = {
				culture_group = yan
				culture_group = upper_raheni
				culture_group = harimari
				culture = goldscale_kobold
				has_reform = xiaken
				has_reform = indep_xiaken
				has_reform = wulin
				has_reform = one_xia
			}
		}
		add_church_power = $amount$
	}
}
