	if = {
		limit = {
			any_owned_province = {

				NOT = {

						trade_goods = coal
					}
				}

					has_winter = mild_winter

			}
		}

				}
			}
			#And it changes the goods to sugar and adds the modifier.
			change_trade_goods = sugar
			add_permanent_province_modifier = {
				name = G58_sugar_cold_resistant_modifier
				duration = -1
			}
