transmutative_path_change_trade_good_trigger = {
	OR = {
		AND = { #Naamjyut
			has_terrain = jungle
			NOT = { trade_goods = tea }
			owner = { culture = naamjyut }
		}
		AND = { #Khom
			trade_goods = cloth
			owner = { culture = khom }
		}
		AND = { #Phonan
			OR = {
				has_terrain = hills
				has_terrain = highlands
				has_terrain = mountain
			}
			NOT = { trade_goods = copper }
			owner = { culture = phonan }
		}
		AND = { #Hinphat
			trade_goods = livestock
			owner = { culture = hinphat }
		}
		AND = { #Binhrung
			trade_goods = fish
			owner = { culture = binhrung }
		}
		AND = { #Bokai
			trade_goods = wool
			owner = { culture = bokai }
		}
		AND = { #Azkare
			trade_goods = cotton
			owner = {
				OR = {
					tag = Y86
					was_tag = Y86
				}
			}
		}
		AND = {  #Thirabnir
			OR = {
				has_terrain = drylands
				has_terrain = desert
			}
			NOT = { trade_goods = salt }
			owner = {
				OR = {
					tag = Y59
					was_tag = Y59
				}
			}
		}
		AND = { #Lo Ngoen
			trade_goods = salt
			owner = {
				OR = {
					tag = Y60
					was_tag = Y60
				}
			}
		}
		AND = { #Khao Elnak
			trade_goods = grain
			owner = {
				OR = {
					tag = Y62
					was_tag = Y62
				}
			}
		}
		AND = { #Antmvura
			has_terrain = jungle
			NOT = { trade_goods = coffee }
			owner = {
				culture = antmvura
			}
		}
	}
}
