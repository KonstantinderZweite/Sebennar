	custom_trigger_tooltip = {
		tooltip = artifice_has_invention_slot_tt
		NOT = {
			check_variable = {
				which = artifice_inventions_currently_used
				which = ArtificeMaxInventions
			}
		}
	}
}

	custom_trigger_tooltip = {

		}

	custom_trigger_tooltip = {
		tooltip = is_not_researching_invention_tt
		NOT = {
			has_country_flag = artifice_research_in_progress
			has_country_flag = artifice_research_menu_open
		}
	}
	if = {
		limit = {
			has_country_modifier = artifice_research_temporarily_disabled
		}
		custom_trigger_tooltip = {
			tooltip = researching_invention_not_disabled_tt
			NOT = {
				has_country_modifier = artifice_research_temporarily_disabled
			}
		}
	}
}

				has_government_attribute = is_magocracy #magocracies are locked to mage only

			}
			has_ruler_flag = is_a_lich #everyone, including eordan elves, becomes mage only if a lich
		}
	}
}
