

#This one or the next one will always be valid
estate_mages_fire_advisor = {
	can_select = {
		employed_advisor = {
			category = ADM
		}
		NOT = {
			employed_advisor = {
				type = court_mage
			}
		}
		NOT = { wyvernheart_is_floodborn_council = yes }
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.5
			always = yes
		}
		modifier = {
			factor = 0.5
			employed_advisor = {
				type = inquisitor
			}
		}
		modifier = {
			factor = 0.5
			employed_advisor = {
				type = natural_scientist
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_influence = {
					estate = estate_mages
					influence = 50
				}
			}
		}
	}
	immediate_effect = {
		hidden_effect = {
			set_country_flag = need_to_fire_adm_advisor
		}
	}
	on_invalid = {
		clr_country_flag = need_to_fire_adm_advisor
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = { estate = estate_mages }
			}
			has_estate_agenda_auto_completion = { estate = estate_mages }
		}
		else = {
			custom_trigger_tooltip = {
				tooltip = fire_adm_advisor_tooltip
				has_country_flag = just_fired_adm_advisor
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_mages
		}
		add_estate_loyalty = {
			estate = estate_mages
			loyalty = 10
		}
		create_advisor = court_mage
		
		clr_country_flag = just_fired_adm_advisor
		clr_country_flag = need_to_fire_adm_advisor
	}
	invalid_trigger = {
		NOT = {
			employed_advisor = {
				category = ADM
			}
		}
		NOT = { has_country_flag = just_fired_adm_advisor }
	}
        failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_mages
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		clr_country_flag = need_to_fire_adm_advisor
    }
}

estate_mages_conquer_damestear = {
	can_select = {
		is_free_or_tributary_trigger = yes
		is_at_war = no
		NOT = { wyvernheart_is_floodborn_council = yes }
		any_neighbor_country = {
			#num_of_cities = 6
			is_free_or_tributary_trigger = yes
			NOT = { is_subject_of = root }
			NOT = { overlord_of = root }
			NOT = { alliance_with = root }
			NOT = { ROOT = { truce_with = prev } }
			NOT = { 
				has_opinion = {
					who = root
					value = 25
				}
			}
			any_owned_province = { #No need for this agenda if covered by a mission
				trade_goods = damestear
				has_discovered = ROOT
			}
			any_owned_province = { #any_neighbor_country includes neighboring tribal land, but any_neighbor_province doesn't - check we have an actual province bordering
				any_neighbor_province = {
					owned_by = ROOT
				}
			}
			ROOT = {
				knows_country = prev
			}
			OR = {
				is_part_of_hre = no
				NOT = { hre_reform_passed = landfriede }
				ROOT = { is_part_of_hre = no }
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.5
			has_non_generic_missions = yes
		}
		modifier = {
			factor = 1.5
			NOT = { damestear = 1 }
		}
		modifier = {
			factor = 1.5
			NOT = { damestear = 3 }
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_neighbor_country = {
			limit = {
				#num_of_cities = 6
				is_free_or_tributary_trigger = yes
				NOT = { is_subject_of = root }
				NOT = { overlord_of = root }
				NOT = { alliance_with = root }
				NOT = { ROOT = { truce_with = prev } }
				NOT = { 
					has_opinion = {
						who = root
						value = 25
					}
				}
				any_owned_province = { #No need for this agenda if covered by a mission
					trade_goods = damestear
					has_discovered = ROOT
				}
				any_owned_province = { #any_neighbor_country includes neighboring tribal land, but any_neighbor_province doesn't - check we have an actual province bordering
					any_neighbor_province = {
						owned_by = ROOT
					}
				}
				ROOT = {
					knows_country = prev
				}
				OR = {
					is_part_of_hre = no
					NOT = { hre_reform_passed = landfriede }
					ROOT = { is_part_of_hre = no }
				}
			}
			random_owned_province = {
				limit = {
					trade_goods = damestear
					has_discovered = ROOT
				}
				save_event_target_as = agenda_province
			}
			random_owned_province = {
				limit = {
					trade_goods = damestear
					has_discovered = ROOT
					any_neighbor_province = {
						owned_by = root
					}
				}
				save_event_target_as = agenda_province
			}
		}
	}
	immediate_effect = {
		event_target:agenda_province = {
			add_claim = root
		}
	}
	on_invalid = {
		if = {
			limit = {
				event_target:agenda_province = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
			}
			event_target:agenda_province = {
				remove_claim = root
			}
		}
		if = {
			limit = {
				event_target:agenda_province = {
					NOT = { trade_goods = damestear }
				}
			}
			event_target:agenda_province = {
				remove_claim = root
			}
		}
	}
	provinces_to_highlight = {
		area_for_scope_province = {
			province_id = event_target:agenda_province
		}
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = { estate = estate_mages }
			}
			has_estate_agenda_auto_completion = { estate = estate_mages }
		}
		else = {
			event_target:agenda_province = {
				country_or_non_sovereign_subject_holds = root
			}
		}
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { is_claim = root }
			NOT = { is_core = root }
			NOT = { owned_by = root }
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = { estate = estate_mages }
		add_estate_loyalty = {
			estate = estate_mages
			loyalty = 20
		}
		add_adm_power = 50
	}
        failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_mages
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		if = {
			limit = {
				event_target:agenda_province = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
			}
			event_target:agenda_province = {
					remove_claim = root
			}
		}
    }
