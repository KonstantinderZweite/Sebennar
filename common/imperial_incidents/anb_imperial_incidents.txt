					who = event_target:imperial_witch_king
					attitude = attitude_rebellious
				}
				is_neighbor_of = event_target:imperial_witch_king
			}
		}
		#Magical Authorities
		modifier = {
			factor = 10
			OR = {
				tag = A85 #Magisterium

				}
				ai_attitude = {
					who = event_target:imperial_witch_king
