
fallen_dwarf = {
	potential = {
		has_country_flag = fallen_invasion_ready
		NOT = { has_global_flag = fallen_dwarf_end }
	}

		custom_trigger_tooltip = {
			tooltip = fallen_start_tooltip
			has_country_flag = fallen_can_start
		}
		has_any_disaster = no
	}
	
	
	can_stop = {
        always = no
	}
	
	progress = {
		modifier = {
			factor = 1
			always = yes
		}
		modifier = {
			factor = 1
			NOT = { stability = 3 }
		}
		modifier = {
			factor = 2
			NOT = { stability = 2 }
		}
		modifier = {
			factor = 3
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 4
			NOT = { stability = 0 }
		}
	}
	
	
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = destroy_obsidian_legion_tooltip
			NOT = {
				war_with = H94
			}
			had_country_flag = {
				flag = obsidian_debug_flag
				days = 90
			}
		}
	}
	

	modifier = {
		land_morale = -0.1
		discipline = -0.05
		free_leader_pool = -1
		reinforce_speed = -0.33
	}

	on_end = diggy_fallen.7
	
	on_monthly = {
		events = {
		}

			1000 = 0
		
		}
	}
