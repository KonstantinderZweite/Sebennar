######################################################################
# Harimari Hardliners.
######################################################################
harimari_hardliners = {

	color = { 229 160 172 }

	area = nation 					# Stays within the nation
	government = any	
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_owner
	
	resilient = no
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = yes
	will_relocate = yes
	
	infantry = 0.6
	cavalry = 0.4
	artillery = 0.0
	
	morale = 1.1
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no		
	handle_action_change_culture = yes	
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 999 # 1 less than blood_lotus_rebel, in case it matters
		modifier = {
			factor = 0
			owner = { NOT = { has_disaster = blood_lotus_rebellion } }
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
	}
	siege_won_effect = {
		add_local_autonomy = 10
		if = {
			limit = {
				OR = {
					has_province_modifier = blood_lotus_headquarter
					has_province_modifier = blood_lotus_guerrillas
				}
			}
			spawn_rebels = {
				type = blood_lotus_rebel
				size = 2
			}
		}
		remove_province_modifier = blood_lotus_headquarter
		remove_province_modifier = blood_lotus_sympathizers
		remove_province_modifier = blood_lotus_guerrillas
		hidden_effect = { add_devastation = 10 }
		# add_nationalism =  10
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "harimari_hardliners_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		every_owned_province = {
			remove_province_modifier = blood_lotus_guerrillas
			remove_province_modifier = blood_lotus_guerrillas_hidden
			remove_province_modifier = blood_lotus_headquarter
			remove_province_modifier = blood_lotus_sympathizers
		}
		add_prestige = -100
		add_stability = -6
		hidden_effect = { set_country_flag = blood_lotus_lost_harimari_hardliner }	
	}
}
