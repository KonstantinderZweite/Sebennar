
######################################################################
# Disgruntled magnates.
######################################################################
magnate_rebels = {

	color = { 179 75 75 }

	area = nation 					# Stays within the nation
	government = republic			# Likes republic
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	unit_transfer = no
	gfx_type = culture_province	
	will_relocate = yes	
	
	resilient = no
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = no
	
	artillery = 0.0
	infantry = 0.6
	cavalry = 0.4
	
	morale = 1.0
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 10
			owner = { 
				has_disaster = magnate_uprising
				has_country_flag = chose_crown
			}
			is_in_capital_area = yes
		}
		modifier = {
			factor = 0
			owner = { 
				NOT = { has_disaster = magnate_uprising }
			}
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		NOT = { local_autonomy = 50 }
	}
	siege_won_effect = {
		add_local_autonomy = 10
		owner = {
			if = {
				limit = { has_estate = estate_burghers }
				change_estate_land_share = {
					estate = estate_nobles
					share = 10.0
					province = ROOT
				}
			}
			else = { # Fallback
				change_estate_land_share = {
					estate = all
					share = 10.0
					province = ROOT
				}
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "magnate_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		every_owned_province = {
			limit = {
				has_rebel_faction = REB
			}
			add_local_autonomy = 20
		}	
		add_prestige = -20
		add_legitimacy = -20
		add_horde_unity = -20
		if = {
			limit = { has_estate = estate_burghers }
			change_estate_land_share = {
				estate = estate_nobles
				share = 10.0
			}
		}
		else = { # Fallback
			change_estate_land_share = {
				estate = all
				share = 10.0
			}
		}
		
		# Anbennar
		if = {
			limit = {
				has_disaster = magnate_uprising
			}
			set_country_flag = just_lost_to_rebels_nl
			hidden_effect = { country_event = { id = magnate_uprising.2 } }
			end_disaster = magnate_uprising
		}
	}
}
