

namespace = seiche

#Random Seiche setup at start
country_event = {
	id = seiche.1
	title = seiche.1.t
	desc = seiche.1.d
	picture = FORT_eventPicture
	
	fire_only_once = yes
	hidden = yes
	
	trigger = {
		always = yes
	}
	
	option = {
		name = seiche.1.a
		ai_chance = { factor = 50 }
		random_list = {
			25 = { set_global_flag = seiche_1 }
			25 = { set_global_flag = seiche_2 }
			25 = { set_global_flag = seiche_3 }
			25 = { set_global_flag = seiche_4 }
		}
		hidden_effect = { set_global_flag = last_federation_crisis }
	}
}

#Disaster - Our province are hitted by a scheiss
country_event = {
	id = seiche.2
	title = seiche.2.t
	desc = seiche.2.d
	picture = FORT_eventPicture
	
	fire_only_once = yes
	major = yes
	major_trigger = { has_country_modifier = lake_federation_member }
	
	mean_time_to_happen = {
		years = 30
	}
	
	trigger = {
		has_country_modifier = lake_federation_member
		NOT = { has_global_flag = federation_crisis }
		NOT = { has_global_flag = federation_crisis_seiche }
		NOT = { has_global_flag = lake_federation_disbanded }
		NOT = { has_global_flag = federation_crisis_goldisland }
		had_global_flag = {
			flag = last_federation_crisis
			days = 9125
		}
		if = {
			limit = { has_global_flag = seiche_1 }
			any_owned_province = { province_group = seiche_1 }
		}
		else_if = {
			limit = { has_global_flag = seiche_2 }
			any_owned_province = { province_group = seiche_2 }
		}
		else_if = {
			limit = { has_global_flag = seiche_3 }
			any_owned_province = { province_group = seiche_3 }
		}
		else_if = {
			limit = { has_global_flag = seiche_4 }
			any_owned_province = { province_group = seiche_4 }
		}
	}
	
	immediate = {
		hidden_effect = { set_global_flag = federation_crisis }
		hidden_effect = { set_global_flag = federation_crisis_seiche }
	}
	
	option = {
		name = seiche.2.a
		ai_chance = { factor = 50 }
		if = {
			limit = { has_global_flag = seiche_1 }
			seiche_1 = {
				add_base_tax = -3
				add_base_manpower = -3
				add_base_production = -3
				add_permanent_province_modifier = {
					name = devasted_province
					duration = -1
				}
				hidden_effect = {
					set_variable = {
						which = seicheFed
						value = 0
					}
				}
			}
			hidden_effect = {
				REB = {
					set_variable = {
						which = seicheFed
						value = 0
					}
					set_variable = {
						which = maxSeicheFed
						value = 63
					}
				}
			}
		}
		else_if = {
			limit = { has_global_flag = seiche_2 }
			seiche_2 = {
				add_base_tax = -3
				add_base_manpower = -3
				add_base_production = -3
				add_permanent_province_modifier = {
					name = devasted_province
					duration = -1
				}
				hidden_effect = {
					set_variable = {
						which = seicheFed
						value = 0
					}
				}
			}
			hidden_effect = {
				REB = {
					set_variable = {
						which = seicheFed
						value = 0
					}
					set_variable = {
						which = maxSeicheFed
						value = 54
					}
				}
			}
		}
		else_if = {
			limit = { has_global_flag = seiche_3 }
			seiche_3 = {
				add_base_tax = -3
				add_base_manpower = -3
				add_base_production = -3
				add_permanent_province_modifier = {
					name = devasted_province
					duration = -1
				}
				hidden_effect = {
					set_variable = {
						which = seicheFed
						value = 0
					}
				}
			}
			hidden_effect = {
				REB = {
					set_variable = {
						which = seicheFed
						value = 0
					}
					set_variable = {
						which = maxSeicheFed
						value = 45
					}
				}
			}
		}
		else_if = {
			limit = { has_global_flag = seiche_4 }
			seiche_4 = {
				add_base_tax = -3
				add_base_manpower = -3
				add_base_production = -3
				add_permanent_province_modifier = {
					name = devasted_province
					duration = -1
				}
				hidden_effect = {
					set_variable = {
						which = seicheFed
						value = 0
					}
				}
			}
			hidden_effect = {
				REB = {
					set_variable = {
						which = seicheFed
						value = 0
					}
					set_variable = {
						which = maxSeicheFed
						value = 54
					}
				}
			}
		}
		hidden_effect = {
			every_known_country = {
				limit = { has_country_modifier = lake_federation_member }
				country_event = { id = seiche.3 }
			}
			country_event = { id = seiche.3 }
		}
	}
}
