	goto = 447
	
	trigger = {
		has_global_flag = corveld_restored #From Decision
		owns = 447
		447 = {
			has_owner_culture = yes
			development = 30
			is_prosperous = yes
			NOT = { num_free_building_slots = 1 }
		}
	}
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.5

		}

	option = { #Push further!

			add_base_tax = 2

	option = { #Good work
        name = corveld_city.3.b
