#Offer administrative support
country_event = {
	id = konwell.2
	title = konwell.2.t
	desc = konwell.2.d
	picture = OVEREXTENSION_eventPicture
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	#Accept autonomous vassalization
	option = {
		name = konwell.2.a
		ai_chance = {
			factor = 100
			#Will never accept if Ryalan temple is elector, emperor, already a vassal, or a different religion
			modifier = {
				factor = 0
				OR = {
					ROOT = { is_elector = yes }
					ROOT = { is_emperor = yes }
					ROOT = { is_vassal = yes }
					NOT = { ROOT = { has_matching_religion = FROM } }
				}
			}
			#More likely if great power
			modifier = {
				factor = 1.5
				A35 = { is_great_power = yes }
			}
			#More likely if opinion > 175
			modifier = {
				factor = 1.5
				has_opinion = {
					who = A35
					value = 175
				}
				NOT = { 
					has_opinion = {
					who = A35
					value = 199
					}
				}
			}
			#More likely if opinion > 199
			modifier = {
				factor = 3
				has_opinion = {
					who = A35
					value = 199
				}
			}
			#More likely if spy network from Konwell > 50
			modifier = {
				factor = 3
				FROM = {
					has_spy_network_in = {
						who = ROOT
						value = 50
					}
				}
			}
			#More likely if Ryalan temple is overextended
			modifier = {
				factor = 1.5
				overextension_percentage = 0.33
			}
			#More likely if Ryalan temple is highly devoted to religious activities
			modifier = {
				factor = 3
				devotion = 90
			}
		}
		add_country_modifier = {
		name = konwell_ryalan_admin_freedom
		duration = -1
		}
		FROM = { 
			country_event = { 
				id = konwell.3 
				days = 30
				random = 5
			} 
		}
	}
	
	#Refuse autonomous vassalization
	option = {
		name = konwell.2.b
		ai_chance = {
			factor = 100
			
			#More likely if tech difference
			modifier = {
				factor = 3
				FROM = {
					tech_difference = 3
				}
			}
			modifier = {
				factor = 5
				FROM = {
					tech_difference = 6
				}
			}
			modifier = {
				factor = 3
				is_great_power = yes
			}
		}
		FROM = {
			country_event = { 
				id = konwell.4 
				days = 30
				random = 5
			} 
		}
	}
}

#AI chooses to take claims from imperial archives or not
country_event = {
	id = konwell.14
	title = konwell.14.t
	desc = konwell.14.d
	picture = SPY_eventPicture
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	trigger = {
		always = yes
	}
	
	immediate = {
		hidden_effect = {
			random_rival_country = {
				limit = {
					any_owned_province = {
						NOT = { is_claim = ROOT }
						NOT = { is_core = ROOT }
						OR = {
							is_capital = no
							owner = {
								NOT = { num_of_cities = 2 }
							}
						}
						any_neighbor_province = {
							owned_by = ROOT
						}
					}
				}
				random_owned_province = {
					limit = {
						NOT = { is_claim = ROOT }
						NOT = { is_core = ROOT }
						OR = {
							is_capital = no
							owner = {
								NOT = { num_of_cities = 2 }
							}
						}
						any_neighbor_province = {
							owned_by = ROOT
						}
					}
					save_event_target_as = archives_claims_province
				}
			}
		}
	}
	#Accept claims
	option = {
		name = konwell.14.a
		ai_chance = {
			factor = 100 
			modifier = {
				factor = 2
				ruler_has_personality = just_personality
			}
			modifier = {
				factor = 2
				ruler_has_personality = righteous_personality
			}
			modifier = {
				factor = 2
				ruler_has_personality = incorruptible_personality
			}
			modifier = {
				factor = 2
				ruler_has_personality = careful_personality
			}
			modifier = {
				factor = 3
				has_opinion = {
					who = FROM
					value = 199
				}
			}
		}
		event_target:archives_claims_province = {
			add_claim = ROOT
		}
		add_trust = {
			who = FROM
			value = 20
			mutual = yes
		}
		#why doesn't an add_favors function exist :(
	}
	#Deny claims 
	option = {
		name = konwell.14.b
		ai_chance = { 
			factor = 100 
			modifier = {
				factor = 2
				ruler_has_personality = intricate_web_weaver_personality
			}
			modifier = {
				factor = 2
				ruler_has_personality = naive_personality
			}
			modifier = {
				factor = 2
				ruler_has_personality = conqueror_personality
			}
			modifier = {
				factor = 2
				ruler_has_personality = expansionist_personality
			}
		}
		add_opinion = {
			who = FROM
			modifier = konwell_emperor_mad
		}
	}
}
