
namespace = marlliande

country_event = {
	id = marlliande.1
	title = marlliande.1.t
	desc = marlliande.1.d
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_reform = ar_callein_reform
		NOT = { has_country_flag = in_marlliande_election }
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = in_marlliande_election
		}
	}
	
	option = {
		name = marlliande.1.a	# Current ruler stays
		trigger = {
			has_regency = no
			NOT = { has_ruler_flag = leader_has_been_pushed_out }
		}
		ai_chance = { 
			factor = 60
			modifier = {
				factor = 0
				NOT = { republican_tradition = 25 }
			}	
			modifier = {
				factor = 0.5
				NOT = { republican_tradition = 50 }
			}				
			modifier = {
				factor = 0.5
				NOT = { republican_tradition = 75 }
			}	
			modifier = {
				factor = 2.0
				republican_tradition = 90
			}		
			
			#Anbennar
			modifier = {
				factor = 2.0
				ruler_has_mage_personality = yes
			}	
		}
		custom_tooltip = remains_ruler
		random_list = {
			10 = {
				change_adm = 1
			}
			10 = {
				change_dip = 1
			}
			10 = {
				change_mil = 1
			}
		}
		random_list = {
			10 = {
				add_adm_power = 50
			}
			10 = {
				add_dip_power = 50
			}
			10 = {
				add_mil_power = 50
			}
		}
		if = {
			limit = { has_country_flag = vilaechi_lord }
			add_faction_influence = {
				faction = vilaechi
				influence = 10
			}
		}
		else_if = {
			limit = { has_country_flag = rosrholych_lord }
			add_faction_influence = {
				faction = rosrholych
				influence = 10
			}
		}
		else_if = {
			limit = { has_country_flag = drominar_lord }
			add_faction_influence = {
				faction = drominar
				influence = 10
			}
		}
		add_scaled_republican_tradition = -10
		hidden_effect = {
			clr_country_flag = in_marlliande_election
		}
	}
	
	#Vilaechi, noblesse oblige clan
	option = {
		name = marlliande.1.b
		trigger = {
			has_regency = no
			NOT = { has_country_flag = vilaechi_lord }
		}
		ai_chance = { factor = 60 }
		define_ruler = {
			change_adm = 2
			change_dip = 0
			change_mil = 0
			max_random_adm = 4
			max_random_dip = 4
			max_random_mil = 4
			random_gender = yes
		}
		add_faction_influence = {
			faction = vilaechi
			influence = 25
		}
		add_adm_power = 50
		hidden_effect = {
			clear_ruler_vampire_flags = yes
			set_country_flag = new_vampire
			ruler_become_vampire = yes
			clr_country_flag = rosrholych_lord
			clr_country_flag = drominar_lord
			set_country_flag = vilaechi_lord
			clr_country_flag = in_marlliande_election
		}
	}
	
	#Rosrholych, diplomatic merchant clan
	option = {
		name = marlliande.1.c
		trigger = {
			has_regency = no
			NOT = { has_country_flag = rosrholych_lord }
		}
		ai_chance = { factor = 60 }
		define_ruler = {
			change_adm = 0
			change_dip = 2
			change_mil = 0
			max_random_adm = 4
			max_random_dip = 4
			max_random_mil = 4
			random_gender = yes
		}
		add_faction_influence = {
			faction = rosrholych
			influence = 25
		}
		add_dip_power = 50
		hidden_effect = {
			clear_ruler_vampire_flags = yes
			set_country_flag = new_vampire
			ruler_become_vampire = yes
			clr_country_flag = vilaechi_lord
			clr_country_flag = drominar_lord
			set_country_flag = rosrholych_lord
			clr_country_flag = in_marlliande_election
		}
	}
	
	#Drominar, militaristic explorer clan
	option = {
		name = marlliande.1.e
		trigger = {
			has_regency = no
			NOT = { has_country_flag = drominar_lord }
		}
		ai_chance = { factor = 60 }
		define_ruler = {
			change_adm = 0
			change_dip = 0
			change_mil = 2
			max_random_adm = 4
			max_random_dip = 4
			max_random_mil = 4
			random_gender = yes
		}
		add_faction_influence = {
			faction = drominar
			influence = 25
		}
		add_mil_power = 50
		hidden_effect = {
			clear_ruler_vampire_flags = yes
			set_country_flag = new_vampire
			ruler_become_vampire = yes
			clr_country_flag = vilaechi_lord
			clr_country_flag = rosrholych_lord
			set_country_flag = drominar_lord
			clr_country_flag = in_marlliande_election
		}
	}
}
