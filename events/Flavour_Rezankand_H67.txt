	desc = rezankand.12.d
	picture = COURT_eventPicture

	mean_time_to_happen = {
		years = 20
		
		modifier = {
			factor = 0.5
			
			has_province_modifier = rezankand_sunny_road
		}
		
		modifier = {
			factor = 0.8
		
			is_capital = yes
		}
		
		modifier = {
			factor = 0.8
			
			NOT = { development = 10 }
		}
		
		modifier = {
			factor = 2
			
			development = 20
		}
		
		modifier = {
			factor = 4
			
			owner = { total_development = 1500 }
		}
		
		modifier = {
			factor = 4
			
			owner = { total_development = 2000 }
		}
	}
	
	trigger = {
		owned_by = H67
		has_port = yes
		colonial_region = colonial_lai_peninsula
		NOT = { is_year = 1700 }
		NOT = { has_province_modifier = rezankand_recent_immigration }
	}

	option = {		
		name = rezankand.12.a
		ai_chance = { factor = 100 }	
		
		if = {
			limit = { is_city = no }
			add_colonysize = 300
		}
		else = {
			add_base_production = 1
			add_base_tax = 1
			add_base_manpower = 1
			if = {
				limit = { owner = { has_country_flag = rezankand_cortez_conquest } }
				add_province_modifier = {
					name = rezankand_recent_immigration 
					duration = 2190
				}
			}
			else = {
				add_province_modifier = {
					name = rezankand_recent_immigration 
					duration = 3650
				}
			}
		}
		add_human_minority_size_effect = yes
	}
}

# Make vassal autonomous vassal
country_event = {
	id = rezankand.14
	title = rezankand.14.t
	desc = rezankand.14.d
	picture = COURT_eventPicture
	
	
	mean_time_to_happen = {
		months = 1
	}
	
	
	
	trigger = {
		tag = H67
		has_country_flag = rezankand_subjugation
		any_subject_country = {
			is_subject_of_type = vassal
			NOT = { has_country_flag = rezankand_subjugated }
		}
	}
	
	immediate = {
		random_subject_country = {
			limit = { 
				NOT = { has_country_flag = rezankand_subjugated } 
				is_subject_of_type = vassal
			}
			set_country_flag = rezankand_subjugated
			save_event_target_as = rezakand_subject
			capital_scope = {
				save_event_target_as = rezakand_subject_capital
			}
		}
	}
	
	goto = rezakand_subject_capital
	
	#We make them autonomous vassal
	option = {		
		name = rezankand.14.a
		ai_chance = { factor = 100 }	
		
		event_target:rezakand_subject = {
			ROOT = {
				create_subject = {
					subject_type = autonomous_vassal
					subject = PREV
				}
			}
			hidden_effect = { set_country_flag = rezankand_subjugated }
		}
	}
	#They stay normal vassal
	option = {		
		name = rezankand.14.b
		ai_chance = { factor = 100 }	
		
		add_stability = -1
		event_target:rezakand_subject = {
			add_trust = {
				who = ROOT 
				value = -20
				mutual = yes
			}
			add_liberty_desire = 65
			hidden_effect = { set_country_flag = rezankand_subjugated }
		}
	}
}
